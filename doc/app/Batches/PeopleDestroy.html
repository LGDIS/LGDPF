def self.execute
  puts(" #{Time.now.to_s} ===== #{self.name} START ===== ")

  # 現在日時より削除予定日時のほうが過去の場合削除する
  Person.destroy_all(["expiry_date < ?", Time.now])

  puts(" #{Time.now.to_s} ===== #{self.name} END ===== ")
end
