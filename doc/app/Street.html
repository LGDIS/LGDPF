def self.hash_for_table(term)
  streets_list = {}
  streets = Street.where("name LIKE '%#{term}%'").order(:code)
  streets.each do |street|
    streets_list[street.code] = {} unless streets_list[street.code]
    streets_list[street.code] = street.name
  end
  return streets_list
end
